Module Board

module Board: sig .. end
exception InvalidCard
type board = {
   players : Player.player list;
   mutable deck : Deck.deck;
   mutable turn : int;
   mutable discarded : Card.card list;
} 
module Mapping: Stdlib.Map.Make(Stdlib.String)
val init_mult_players : int -> string list -> Player.player list
val initialize_board : int -> string list -> board

initialize_board n player_names creates a board of n players with names names.

val distribute_cards_to_players : board -> unit

distribute_cards_to_players board distributes 5 cards to all the players and an additional 2 cards to the first player in board.

val get_current_turn : board -> int

get_current_turn board returns the current turn of board.

val get_players : board -> Player.player list

get_players board is a list of all the players in board.

val get_current_player : board -> string

get_current_player board is the name of the player whose turn it is.

val get_player_names : board -> string list

get_player_names board is a list of all the names of the players in board.

val get_player_from_name : board -> string -> Player.player

get_player_from_name is the player corresponding to the name name for board board

val get_card_value : int -> Card.card list -> int

get_card_value id card_list is the value of the card with id id in card_list.

val draw_new_cards : board -> bool -> unit

draw_new_cards board mode adds up to 2 cards to the current player's hand such that they have no more than 7 cards, and adds 5 cards to the current player's hand if they have 0 cards and mode is true.

val increment_turn : board -> unit

increment_turn board switches the turn to the next person in board.

val add_card_to_pile : board -> int -> unit

add_card_to_pile board id adds the card with id id from the current player's hand to their pile, and raises InvalidCard if that card is not in their hand.

val discard_card_from_hand : board -> int -> unit

discard_card_from_hand board id adds the card with id id from the current player's hand to the discarded list, and raises InvalidCard if that card is not in their hand.

val transfer_card : int -> Player.player -> Player.player -> unit

transfer_card transfers card with id id from player1 pile to player2 pile. Raises InvalidCard if player1 does not have that card in their pile.

val transfer_set : 'a -> Card.color -> Player.player -> Player.player -> unit

transfer_set transfers the set of color from from_player and puts it in the pile of to_player, for the board in this game. Property cards are prioritized over wildcards. Requires: from_player owns the set of color

val check_win : board -> bool

check_win board is true if the current player has at least 3 complete sets and has won the game, and is false otherwise.

val collect_cards : Card.card list -> Card.card list Mapping.t

collect_cards card_list is a mapping of the cards in card_list to their card type.

val extract_color : Card.card -> Card.color

extract_color is the color of the property card card.

val sorted : Card.card list -> Card.card list

sorted sorts plist in alphabetical order by colors.

val print_card_list : Card.card list -> unit

print_card_list card_list prints the sorted card_list.

val print_current_player_hand : board -> unit

print_current_player_hand board prints out the cards in the current player's hand in board.

val print_current_player_pile : board -> unit

print_current_player_pile board prints out the cards in the current player's pile in board.

val print_pile_of_player : board -> string -> unit

print_pile_of_player board player_name prints out the pile of the player whose name is player_name in board.
