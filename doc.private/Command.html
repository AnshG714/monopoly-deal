Module Command

module Command: sig .. end
type command =
| Draw
| Play of int
| ViewPile
| ViewHand
| Pile of string
| Discard of int
| Pass
| Quit 
exception Malformed of string
exception Empty
val id_validator : string list -> string -> command

id_validator sl keyword creates command keyword if sl is a list of valid integers. Raises Malformed otherwise.

val parse_helper : string list -> command

parse_helper strlist parses strlist into a command h where h is the first element of strlist.

val parse : string -> command

parse str parses str into a command if possible. Raises Empty if str is empty. Raises Malformed if str is an invalid command.
